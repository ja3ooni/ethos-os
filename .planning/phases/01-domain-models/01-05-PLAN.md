```python
#!/usr/bin/env python
"""Seed script for development."""
import sys
from pathlib import Path

# Add ethos_os to path
sys.path.insert(0, str(Path(__file__).parent))

from ethos_os.db import get_session
from ethos_os.repositories import PortfolioRepository, ProgramRepository, ProjectRepository, SprintRepository

def seed(session):
    # Create portfolio
    portfolio = PortfolioRepository(session).create({
        'name': 'Demo Corp',
        'strategic_intent': 'Revolutionize human-agent collaboration',
        'owner_id': None
    })
    
    # Create program
    program = ProgramRepository(session).create({
        'name': 'EthosOS',
        'portfolio_id': portfolio.id,
        'description': 'The flagship initiative',
        'owner_id': None
    })
    
    # Create project
    project = ProjectRepository(session).create({
        'name': 'v0.1 MVP',
        'program_id': program.id,
        'intent': 'Ship core initiative hierarchy',
        'success_metric': 'Portfolio → subtask traversal works',
        'scope': 'Domain models, working memory, no API',
        'boundaries': 'No API, no heartbeat, no gates',
        'timeline': '{"start": "2026-04-24", "end": "2026-05-01"}',
        'budget': 1000000,  # $10,000
        'prd_status': 'approved',  # Bypass approval for seed
        'owner_id': None
    })
    
    # Create sprint
    sprint = SprintRepository(session).create({
        'name': 'Sprint 1',
        'project_id': project.id,
        'goal': 'Get domain models working',
        'start_date': '2026-04-24',
        'end_date': '2026-05-01',
        'capacity_hours': 80,
        'status': 'active',
        'owner_id': None
    })
    
    print(f"Created: {portfolio.name} → {program.name} → {project.name} → {sprint.name}")
    return portfolio, program, project, sprint

if __name__ == '__main__':
    session = next(get_session())
    seed(session)
    session.commit()
    print("Seed complete!")
```

```python
# verify_hierarchy.py
import sys
sys.path.insert(0, '.')
from ethos_os.db import get_session
from ethos_os.repositories import ProjectRepository, SprintRepository, TaskRepository

def verify():
    s = next(get_session())
    repo = ProjectRepository(s)
    # Query projects with lineage
    projects = repo.list({})
    for p in projects:
        lineage = repo.get_with_lineage(p.id)
        print(f"Lineage: {lineage}")
    
    # Verify task validation
    try:
        TaskRepository(s).create({'name': 'Bad Task', 'sprint_id': None, 'prd_scope_item_id': None})
        print("ERROR: Task should require prd_scope_item_id")
    except Exception as e:
        print(f"Correctly rejected: {e}")

if __name__ == '__main__':
    verify()
```
